17:  // Helper function to extract high-fidelity fields from Google Places API (New) components
18:  function parseGooglePlace(place: any) {
19:    const name = place.displayName?.text || "";
20:    const formattedAddress = place.formattedAddress || "";
21:    const components = place.addressComponents || [];
22:
23:    let state = "";
24:    let city = "";
25:    let postcode = "";
26:    let street_address = "";
27:
28:    const findComponent = (types: string[]) => {
29:      return components.find((c: any) => c && Array.isArray(c.types) && c.types.some((t: string) => types.includes(t)))?.longText || "";
30:    };
31:
32:    postcode = findComponent(["postal_code"]);
33:    state = findComponent(["administrative_area_level_1"]);
34:
35:    // Choose city/township
36:    city = findComponent(["locality"]) || 
37:           findComponent(["sublocality_level_1"]) || 
38:           findComponent(["administrative_area_level_2"]) ||
39:           "Kuala Lumpur";
40:
41:    // Build street address from components
42:    const streetNo = findComponent(["street_number"]);
43:    const route = findComponent(["route"]);
44:    const neighborhood = findComponent(["neighborhood"]) || findComponent(["sublocality"]);
45:
46:    const parts: string[] = [];
47:    if (streetNo) parts.push(streetNo);
48:    if (route) parts.push(route);
49:    if (neighborhood) parts.push(neighborhood);
50:
51:    street_address = parts.join(", ");
52:
53:    // If street address is too short, generate from formattedAddress
54:    if (!street_address || street_address.length < 5) {
55:      const rawParts = formattedAddress.split(',').map((p: string) => p.trim());
56:      const filtered = rawParts.filter((p: string) => {
57:        const l = p.toLowerCase();
58:        if (l === 'malaysia') return false;
59:        if (postcode && l.includes(postcode)) return false;
60:        if (state && (l.includes(state.toLowerCase()) || state.toLowerCase().includes(l))) return false;
61:        if (city && (l.includes(city.toLowerCase()) || city.toLowerCase().includes(l))) return false;
62:        if (name && (l.includes(name.toLowerCase()) || name.toLowerCase().includes(l))) return false;
63:        return true;
64:      });
65:      street_address = filtered.join(', ');
66:    }
67:
68:    // Default fallback to stripped address
69:    if (!street_address) {
70:      street_address = formattedAddress.replace(/,\s*Malaysia$/i, "");
71:    }
72:
73:    const latitude = place.location?.latitude || 3.1390;
74:    const longitude = place.location?.longitude || 101.6869;
75:
76:    return {
77:      building_name: name || "N/A",
78:      street_address: street_address,
79:      postcode: postcode || "50000",
80:      city: city,
81:      state: state,
82:      country: "Malaysia",
83:      latitude: latitude,
84:      longitude: longitude
85:    };
86:  }
